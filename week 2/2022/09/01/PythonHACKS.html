def getValue(dic, key):
    if (key in dic):
        return str(dic[key])

def showValue(dic, key):
    if (key in dic):
        print(key + " : " + getValue(dic, key))

# len(InfoDB) returns length of InfoDB as an integer
# range command returns a sequence of numbers, starting from 0 by default and ends (in this case) with the length of InfoDB
for x in range(len(InfoDB)):
    print()
    # getting the dictionary stored at index position x, assigned to variable "item"
    item = InfoDB[x]
    # printing out every key and value in the dictionary using y as the index
    for y in range(len(item)):
        print(list(item)[y] +" : " + str(list(item.values())[y]))

Iteration through InfoDB Using "Recursion" 

def recursive_loop(n):
    
    # setting an if statement that "if n is less than the length of InfoDB, follow this code:"
    if n < len(InfoDB):
        # assigning dictionary to the variable DB
        DB = InfoDB[n]
        # printing out all the keys and values in the DB dictionary
        print()
        for key in DB:
            print(str(key) + " : " + str(DB[key]))
        # recursion occurs here, the function is called again but n is added and reassigned to a value one greater
        recursive_loop(n + 1)

def AddToDB(x, key):
    print ("Input the " + key + " for the Person")
    value = input()
    x[key] = value

def AddToDBList(x, key):
    valueList = []
    while True :
        print ("Input the " + key + " for the Person")
        value = input()
        valueList.append(value)
        print("Do you want to add more values for " + key + ". Enter 'Y' for yes and 'N' for no.")
        check = input()
        if check == "N":
            break
    x[key] = valueList
