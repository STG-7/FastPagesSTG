Iteration through InfoDB using "For Loop" with Indexes 

# len(InfoDB) returns length of InfoDB as an integer
# range command returns a sequence of numbers, starting from 0 by default and ends (in this case) with the length of InfoDB
for x in range(len(InfoDB)):
  
  # printing out every key and value in the dictionary using x as the index
  print(list(InfoDB)[x] +" : " + str(list(InfoDB.values())[x]))

Iterating through InfoDB using "While Loop" 

keynumber = 0
valuenumber = 0

# setting condition that while keynumber is less than 0 follow this code:
while keynumber < len(InfoDB) :

    # setting local key and val variables to key and values form InfoDB dictionary with the indexes of keynumber nad valuenumber
    key = list(InfoDB)[keynumber]
    val = list(InfoDB.values())[valuenumber] 
    
    # printing out key and val variables with str() because there are lists in InfoDB 
    print(str(key) + ": " + str(val))
    
    # adding and reassigning keynumber and valuenumber so that the code will go on until all of the content in InfoDB is printed out
    keynumber += 1
    valuenumber += 1

Iteration through InfoDB Using "Recursion" 

def recursive_loop(n):
    
    # setting an if statement that "if" n is less than the length of InfoDB, follow this code:
    if n < len(InfoDB):

        # printing out the key and value in the dictionary that has the index "n"
        print(list(InfoDB)[n] +" : " + str(list(InfoDB.values())[n]))
        
        # recursion occurs here, the function is called again but n is added and reassigned to a value one greater
        recursive_loop(n + 1)
